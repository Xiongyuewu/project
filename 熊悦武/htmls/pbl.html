暂停添加 打开文档

首先要引入jquery建议版本是jquery-1.12.2.min.js;因为原生开发版本就是这个版本

在引入我分装的pbl.js

//下面是需要根据自己的需求来配置瀑布流基本信息

var opts = {

imgName : [1,2,3,4,5,6,7,8],//图片名字

arrHeight : [100,200,180,110,130,150],//装载图片容器的高度

width : 110,//装载图片容器的宽度

space : 5,//装载图片容器两两之间的间距

margin : 5,//图片与所在容器间的间距

id:"#container",//父级容器

background:"#000",//装载容器的背景颜色，默认为白色

ID:"add",//查看图片的id选择器

}

zzmpbl(opts);//启动瀑布流
